// Nodemon
nodemon index.js --PORT 8080 --MODE "FORK"
nodemon index.js --PORT 8081 --MODE "CLUSTER"

// Forever
forever start ./index.js --watch -- -p 8080 -m "FORK"
forever start ./index.js --watch -- -p 8081 -m "CLUSTER"
forever list
forever stopall

// PM2
pm2 start index.js --name "fork-server" --watch -- 8080
pm2 start index.js --name "cluster-server" -i max --watch -- 8081
pm2 list
pm2 stop all
pm2 delete all

// Sistema
tasklist /fi "imagename eq node.exe"
taskkill /pid <PID> /f
